**Date management — URL-synced:**
```javascript
const todayStr = () => new Date().toISOString().slice(0, 10)
const selectedDate = computed(() => route.query.date ?? todayStr())

function navigateToDate(dateStr) {
  router.push({ path: '/timeline', query: { date: dateStr } })
}
```

**Data fetching:**
```javascript
async function fetchTimeline() {
  loading.value = true
  error.value = null
  try {
    const res = await fetch(`/api/timeline?date=${selectedDate.value}`)
    if (!res.ok) throw new Error(`HTTP ${res.status}`)
    timelineData.value = await res.json()
    initVisibility(timelineData.value.projects)
  } catch (e) {
    error.value = e.message
  } finally {
    loading.value = false
  }
}
```

**Project visibility management:**
```javascript
function initVisibility(projects) {
  // Only add new projects, preserve existing visibility state
  for (const p of projects) {
    if (!projectVisibility.value.has(p.projectId)) {
      projectVisibility.value.set(p.projectId, true)
    }
  }
}

function toggleProject(projectId) {
  const current = projectVisibility.value.get(projectId)
  projectVisibility.value.set(projectId, !current)
  // Force reactivity by creating new Map
  projectVisibility.value = new Map(projectVisibility.value)
}
```

**Project color assignment — djb2 hash:**
```javascript
const COLOR_PALETTE = [
  '#4e9af1', '#f4a523', '#2ebd6b', '#e05c5c', '#a87fe0',
  '#00c4bc', '#f06292', '#8bc34a', '#ff8f00', '#78909c',
]

function projectColor(projectPath) {
  let hash = 5381
  for (const char of projectPath) hash = (hash * 33) ^ char.charCodeAt(0)
  return COLOR_PALETTE[Math.abs(hash) % COLOR_PALETTE.length]
}
```

**Computed: colorized and filtered projects:**
```javascript
const colorizedProjects = computed(() => {
  if (!timelineData.value?.projects) return []
  return timelineData.value.projects.map(p => ({
    ...p,
    color: projectColor(p.projectPath),
  }))
})

const visibleProjects = computed(() =>
  colorizedProjects.value.filter(p => projectVisibility.value.get(p.projectId) !== false)
)

const legendItems = computed(() =>
  colorizedProjects.value.map(p => ({ displayName: p.displayName, color: p.color }))
)
```

**Import trigger:**
```javascript
async function triggerImport() {
  if (importRunning.value) return
  importRunning.value = true
  try {
    const res = await fetch('/api/import', { method: 'POST' })
    if (res.status === 409) return  // already running
    if (!res.ok) throw new Error('Import failed')
    await fetchTimeline()  // auto-refresh
  } catch (e) {
    error.value = e.message
  } finally {
    importRunning.value = false
  }
}
```

**Lifecycle and watchers:**
```javascript
// Fetch on mount
onMounted(fetchTimeline)

// Re-fetch when date changes via URL
watch(() => route.query.date, () => fetchTimeline())
```

**Template:**
```vue
<template>
  <div class="timeline-page">
    <!-- Toolbar: date navigation + import -->
    <TimelineToolbar
      :date="selectedDate"
      :import-running="importRunning"
      @navigate="navigateToDate"
      @import="triggerImport"
    />

    <!-- Main content area -->
    <div class="timeline-content">
      <!-- Loading state -->
      <div v-if="loading" class="timeline-loading">
        Loading timeline...
      </div>

      <!-- Error state -->
      <div v-else-if="error" class="timeline-error">
        <p>Error loading timeline: {{ error }}</p>
        <AppButton variant="secondary" size="sm" @click="fetchTimeline">Retry</AppButton>
      </div>

      <!-- Empty state -->
      <div v-else-if="!timelineData?.projects?.length" class="timeline-empty">
        <h2>No sessions found for {{ selectedDate }}</h2>
        <p>Try importing your Claude Code transcripts, or navigate to a different date.</p>
        <AppButton
          variant="primary"
          :loading="importRunning"
          @click="triggerImport"
        >
          Import Transcripts
        </AppButton>
      </div>

      <!-- Timeline with data -->
      <template v-else>
        <!-- Project filter -->
        <div class="filter-bar" v-if="colorizedProjects.length > 1">
          <span class="filter-label">Projects:</span>
          <label
            v-for="project in colorizedProjects"
            :key="project.projectId"
            class="filter-item"
          >
            <AppCheckbox
              :model-value="projectVisibility.get(project.projectId) !== false"
              @update:model-value="toggleProject(project.projectId)"
              :label="project.displayName"
            />
          </label>
        </div>

        <!-- Legend -->
        <GanttLegend :projects="legendItems" />

        <!-- Gantt chart -->
        <GanttChart
          :projects="visibleProjects"
          :date="selectedDate"
        />
      </template>
    </div>
  </div>
</template>
```
